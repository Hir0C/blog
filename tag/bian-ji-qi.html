    --with-features=huge 一定要加

        ./configure --prefix=/usr/local/vim74 \
     **--with-features=huge** \
     --enable-pythoninterp --enable-perlinterp \
     --enable-rubyinterp --enable-luainterp \
     --enable-multibyte --enable-sniff --enable-fontset \
     --disable-selinux --enable-cscope
