简介

github上找到个go写的通过http POST请求发送邮件的代码。
少有的根据文档顺利安装成功并运行的代码。

链接在此

配置

{
    "debug":true,
    "http":{
        "enable":true,
        "listen":"0.0.0.0:6666"
    },
    "rpc":{
        "enable":false,
        "listen":"0.0.0.0:77777"
    },
    "mail":{
        "enable" : true,
        "sendConcurrent" : 5,
        "maxQueueSize" : 100,
        "fromUser" : "user",  //随便写
        "mailServerHost" : "mailserver.com",
        "mailServerPort" : 25,
        "mailServerAccount" : "alert@mailserver.com",  //邮件发件账号名
        "mailServerPasswd" : ""  //没验证可以留空
    }
}
