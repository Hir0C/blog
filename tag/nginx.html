配置nginx支持mp4视频播放

日 12 六月 2016

Filed under 运维

Tags nginx

下载地址

nginx-1.11.1
zlib-1.2.8
nginx_mod_h264_streaming-2.2.7
pcre

修改

需要先修改nginx_mod_h264_streaming一处源码，否则编译会报错。158~161行注释掉，修改后代码如下：

      /* TODO: Win32
      if (r->zero_in_uri)
      {
        return NGX_DECLINED;
      }*/

编译

./configure --prefix=/usr/local/nginx-1.11.1 --user=www --group=www --with-threads --with-http_flv_module --with-http_mp4_module --with-http_gunzip_module --with-http_gzip_static_module --without-mail_pop3_module --without-mail_imap_module --without-mail_smtp_module --with-pcre=/tmp/yy/pcre-8.38 --with-zlib --with-http_ssl_module --add-module=/tmp/yy/nginx_mod_h264_streaming-2.2.7 --with-cc-opt='-O3'

配置

user  www www;
worker_processes  2;

events {
    worker_connections  10240;
}

worker_rlimit_nofile 10240;

http {
    include       mime.types;
    default_type  application/octet-stream;

    sendfile        on;
    tcp_nopush     on;
    tcp_nodelay on;

    keepalive_timeout  65;

    gzip  on;

    # 配置缓存
    proxy_temp_path /home/temp_dir;
    proxy_cache_path /home/cache levels=1:2 keys_zone=cache_one:200m inactive=15d max_size=3g;

    gzip_min_length   1k;
    gzip_buffers   4 8k;
    gzip_http_version  1.1;

    server {
        listen       80;
        server_name  localhost;
        root   /usr/local/nginx/html/;
        index  index.html;

        # 开启基本密码认证
        auth_basic "Restricted";
        auth_basic_user_file htpasswd;

        location / {
            root   /usr/local/nginx/html/;
            index  index.html;
            proxy_ignore_headers X-Accel-Expires Expires Cache-Control Set-Cookie;
        }

        # 添加缓存
        location ~* ^.+\.(js|ico|gif|jpg|jpeg|png|html|htm)$ {
            log_not_found off;
            access_log off;
            expires 7d;
            proxy_ignore_headers X-Accel-Expires Expires Cache-Control Set-Cookie;
        }

        # 添加对mp4文件支持
        location ~ \.mp4$ {
            mp4;
            proxy_ignore_headers X-Accel-Expires Expires Cache-Control Set-Cookie;
        }

        error_page   404  /;

        # 自定义404返回页面
        location = / {
            root   /usr/local/nginx/html/;
        }

        error_page   500 502 503 504  /50x.html;
        location = /50x.html {
            root   html;
        }
    }
}

    }

二、if的or条件

location / {
    set $flag 0;
    if ($query_string ~ "xxx"){
            set $flag 1;
            proxy_pass http://s1;
    }
    if ($query_string ~ "OOO"){
            set $flag 1;
            proxy_pass http://s1;
    }
    if ($flag = "0"){
            return 200;
    }
}
